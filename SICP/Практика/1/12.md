```scheme
(define (! n)
  (define (fact-iter acc n)
	  (fact-iter (* n acc) (- acc 1))
  )
  (fact-iter 1 n)
)
(define (combination n k)
	(/ (! n) (* (! k) (! (- n k))
)
(define (pascal element)
	(define collumn (calculate-pascal-collumn element 1))
	(combination (reminder element collumn) collumn)
)
(define (calculate-pascal-collumn x next-guess)
	(if (<= x 0)
		next-guess
		(calculate-pascal-collumn (- x next-guess) (+ next-guess 1))
	)
)
```
